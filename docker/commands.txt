SETUP :
    1/ cd Reactor_Runs
    2/ run "./compress" if ChemDBPublic and/or ReactorCodes have been updated

BUILD :
    1/ cd Titan-APSIS
    2/ docker build -t ppernot1/reactorui -f ReactorUI/docker/Dockerfile .
    3/ docker push ppernot1/reactorui

RUN :
   UI :
       1/ cd Projects
       2/ docker run -d -p 3838:3838 --mount type=bind,source="$(pwd)",target=/Projects --name reactorui --user $(id -u):$(id -g) ppernot1/reactorui
       3/ access http://localhost:3838 in browser

   BASH :
       * docker run -ti ppernot1/reactorui bash
